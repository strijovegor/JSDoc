/**
считает площадь треугольника

@param {side} a - storona treugolnik
@param {side} b - storona treugolnik
@param {side} c - storona treugolnik

*/

function SquareGeron (a, b, c) {
	let p = (a + b + c) / 2;
	p = p * (p - a) * (p - b) * (p - c);
	p = Math.sqrt(p);
	return p;
}
